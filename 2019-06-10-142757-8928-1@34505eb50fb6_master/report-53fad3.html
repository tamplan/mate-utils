376static gboolean
377read_gzip_header (GInputStream *is,
378 time_t *modification_time)
379{
380 guchar buffer[GZIP_HEADER_SIZE10];
381 gssize bytes, to_skip;
382 guint mode;
383 guint flags;
384
385 bytes = g_input_stream_read (is, buffer, GZIP_HEADER_SIZE10,
386 NULL((void*)0), NULL((void*)0));
387 if (bytes == -1) {
388 return FALSE(0);
389 }
390
391 if (bytes != GZIP_HEADER_SIZE10)
392 return FALSE(0);
393
394 if (buffer[0] != GZIP_MAGIC_10x1f || buffer[1] != GZIP_MAGIC_20x8b)
395 return FALSE(0);
396

399 return FALSE(0);
400

402

416 bytes = g_input_stream_skip (is, to_skip, NULL((void*)0), NULL((void*)0));
417 if (bytes != to_skip) {

506 Z_BUFSIZE16384,
507 NULL((void*)0), NULL((void*)0));
508 if (count == -1) {
509 if (zstream->avail_out == num_bytes) {
510 return FALSE(0);
511 }
512 gz->last_str_result = FALSE(0);
513 } else {
514 zstream->next_in = gz->buffer;
515 zstream->avail_in = count;
516 }
517
518 return TRUE(!(0));
519}

521static gboolean
522result_from_z_result (int z_result)
523{
524 switch (z_result) {
525 case Z_OK0:
526 case Z_STREAM_END1:
527 return TRUE(!(0));
528 case Z_DATA_ERROR(-3):
529 return FALSE(0);
530 default:
531 return FALSE(0);
532 }
533}

535static gboolean
536gz_handle_read (GZHandle *gz,
537 guchar *buffer,
538 gsize num_bytes,
539 gsize * bytes_read)
540{
541 z_stream *zstream;
542 gboolean res;
543 int z_result;
544
545 *bytes_read = 0;
546 zstream = &gz->zstream;
547
548 if (gz->last_z_result != Z_OK0) {
549 if (gz->last_z_result == Z_STREAM_END1) {
550 *bytes_read = 0;
551 return TRUE(!(0));
552 } else {
553 return result_from_z_result (gz->last_z_result);
554 }
555 } else if (gz->last_str_result == FALSE(0)) {
556 return FALSE(0);
557 }
558
559 zstream->next_out = buffer;
560 zstream->avail_out = num_bytes;
561
562 while (zstream->avail_out != 0) {
563 res = fill_buffer (gz, num_bytes);
564
565 if (!res) {
566 return res;
567 }
568
569 z_result = inflate (zstream, Z_NO_FLUSH0);
570 if (z_result == Z_STREAM_END1) {
571 gz->last_z_result = z_result;
572 break;
573 } else if (z_result != Z_OK0) {
574 gz->last_z_result = z_result;
575 }
576
577 if (gz->last_z_result != Z_OK0 && zstream->avail_out == num_bytes) {
578 return result_from_z_result (gz->last_z_result);
579 }
580 }
581
582 *bytes_read = num_bytes - zstream->avail_out;
583
584 return TRUE(!(0));
585}

600static gboolean
601log_load (GIOSchedulerJob *io_job,
602 GCancellable *cancellable,
603 gpointer user_data)
604{
605 /* this runs in a separate i/o thread */
606 LoadJob *job = user_data;
607 LogviewLog *log = job->log;
608 GFile *f = log->priv->file;
609 GFileInfo *info;
610 GInputStream *is;
611 const char *peeked_buffer;
612 const char * parse_data[2];
613 GSList *days;
614 const char *content_type;
615 GFileType type;
616 GError *err = NULL((void*)0);
617 GTimeVal timeval;
618 gboolean is_archive, can_read;
619
620 info = g_file_query_info (f,
621 G_FILE_ATTRIBUTE_ACCESS_CAN_READ"access::can-read" ","
622 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE"standard::content-type" ","
623 G_FILE_ATTRIBUTE_STANDARD_DISPLAY_NAME"standard::display-name" ","
624 G_FILE_ATTRIBUTE_STANDARD_TYPE"standard::type" ","
625 G_FILE_ATTRIBUTE_STANDARD_SIZE"standard::size" ","
626 G_FILE_ATTRIBUTE_TIME_MODIFIED"time::modified" ",",
627 0, NULL((void*)0), &err);
628 if (err) {
629 if (err->code == G_IO_ERROR_PERMISSION_DENIED) {
630 /* TODO: PolicyKit integration */
631 }
632 goto out;
633 }
634
635 can_read = g_file_info_get_attribute_boolean (info,
636 G_FILE_ATTRIBUTE_ACCESS_CAN_READ"access::can-read");
637 if (!can_read) {
638 /* TODO: PolicyKit integration */
639 err = g_error_new_literal (LOGVIEW_ERROR_QUARKg_quark_from_static_string ("logview-error"), LOGVIEW_ERROR_PERMISSION_DENIED,
640 _("You don't have enough permissions to read the file.")gettext ("You don't have enough permissions to read the file."
));
641 g_object_unref (info);
642
643 goto out;
644 }
645
646 type = g_file_info_get_file_type (info);
647 content_type = g_file_info_get_content_type (info);
648
649 is_archive = g_content_type_equals (content_type, "application/x-gzip");
650
651 if (type != (G_FILE_TYPE_REGULAR || G_FILE_TYPE_SYMBOLIC_LINK) ||
652 (!g_content_type_is_a (content_type, "text/plain") && !is_archive))
653 {
654 err = g_error_new_literal (LOGVIEW_ERROR_QUARKg_quark_from_static_string ("logview-error"), LOGVIEW_ERROR_NOT_A_LOG,
655 _("The file is not a regular file or is not a text file.")gettext ("The file is not a regular file or is not a text file."
));
656 g_object_unref (info);
657
658 goto out;
659 }
660
661 log->priv->file_size = g_file_info_get_size (info);
662 g_file_info_get_modification_time (info, &timeval);
663 log->priv->file_time = timeval.tv_sec;
664 log->priv->display_name = g_strdup (g_file_info_get_display_name (info));
665
666 g_object_unref (info);
667
668 /* initialize the stream */
669 is = G_INPUT_STREAM (g_file_read (f, NULL, &err))((((GInputStream*) g_type_check_instance_cast ((GTypeInstance
*) ((g_file_read (f, ((void*)0), &err))), ((g_input_stream_get_type
()))))));
670
671 if (err) {
672 if (err->code == G_IO_ERROR_PERMISSION_DENIED) {
673 /* TODO: PolicyKit integration */
674 }
675
676 goto out;
677 }
678
679 if (is_archive) {
680#ifdef HAVE_ZLIB1
681 GZHandle *gz;
682 gboolean res;
683 guchar * buffer;
684 gsize bytes_read;
685 GInputStream *real_is;
686 time_t mtime; /* seconds */
687
688 /* this also skips the header from |is| */
689 res = read_gzip_header (is, &mtime);
690
691 if (!res) {
692 g_object_unref (is);
693
694 err = create_zlib_error ();
695 goto out;
696 }
697
698 log->priv->file_time = mtime;
699
700 gz = gz_handle_new (f, is);
701 res = gz_handle_init (gz);
702
703 if (!res) {
704 g_object_unref (is);
705 gz_handle_free (gz);
706
707 err = create_zlib_error ();
708 goto out;
709 }
710
711 real_is = g_memory_input_stream_new ();
712
713 do {
714 buffer = g_malloc (1024);
715 res = gz_handle_read (gz, buffer, 1024, &bytes_read);
716 g_memory_input_stream_add_data (G_MEMORY_INPUT_STREAM (real_is)((((GMemoryInputStream*) g_type_check_instance_cast ((GTypeInstance
*) ((real_is)), ((g_memory_input_stream_get_type ())))))),
717 buffer, bytes_read, g_free);
718 } while (res == TRUE(!(0)) && bytes_read > 0);
719
720 if (!res) {
721 gz_handle_free (gz);
722 g_object_unref (real_is);
723 g_object_unref (is);
724
725 err = create_zlib_error ();
726 goto out;
727 }
728
729 g_object_unref (is);
730 is = real_is;
731
732 gz_handle_free (gz);
733#else /* HAVE_ZLIB */
734 g_object_unref (is);
735
736 err = g_error_new_literal (LOGVIEW_ERROR_QUARKg_quark_from_static_string ("logview-error"), LOGVIEW_ERROR_NOT_SUPPORTED,
737 _("This version of System Log does not support GZipped logs.")gettext ("This version of System Log does not support GZipped logs."
));
738 goto out;
739#endif /* HAVE_ZLIB */
740 }
741
742 log->priv->stream = g_data_input_stream_new (is);
743
744 /* sniff into the stream for a timestamped line */
745 g_buffered_input_stream_fill (G_BUFFERED_INPUT_STREAM (log->priv->stream)((((GBufferedInputStream*) g_type_check_instance_cast ((GTypeInstance
*) ((log->priv->stream)), ((g_buffered_input_stream_get_type
())))))),
746 (gssize) g_buffered_input_stream_get_buffer_size (G_BUFFERED_INPUT_STREAM (log->priv->stream)((((GBufferedInputStream*) g_type_check_instance_cast ((GTypeInstance
*) ((log->priv->stream)), ((g_buffered_input_stream_get_type
()))))))),
747 NULL((void*)0), &err);
748 if (err == NULL((void*)0)) {
749 peeked_buffer = g_buffered_input_stream_peek_buffer
750 (G_BUFFERED_INPUT_STREAM (log->priv->stream)((((GBufferedInputStream*) g_type_check_instance_cast ((GTypeInstance
*) ((log->priv->stream)), ((g_buffered_input_stream_get_type
())))))), NULL((void*)0));
751 parse_data[0] = peeked_buffer;
752 parse_data[1] = NULL((void*)0);
753
754 if ((days = log_read_dates (parse_data, time (NULL((void*)0)))) != NULL((void*)0)) {
755 log->priv->has_days = TRUE(!(0));
756 g_slist_free_full (days, (GDestroyNotify) logview_utils_day_free);
757 } else {
758 log->priv->has_days = FALSE(0);
759 }
760 } else {
761 log->priv->has_days = FALSE(0);
762 g_clear_error (&err);
763 }
764
765 g_object_unref (is);
766
767out:
768 if (err) {
769 job->err = err;
770 }
771
772 g_io_scheduler_job_send_to_mainloop_async (io_job,
773 log_load_done,
774 job, NULL((void*)0));
775 return FALSE(0);
776}
